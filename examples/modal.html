Simple example of a dynamic modal with a title, body, button and event listeners.

Source of exampleCopy to clipboard


const myModal1 = new AvalynxModal('modal1', {
    title: 'My dynamic modal title',
    body: 'My dynamic modal content',
    buttons: [
        {
            label: 'Close',
            class: 'btn-secondary',
            onClick: () => myModal1.close()
        }
    ],
    onFullscreenToggled: (instance) => {
        console.log('Fullscreen toggled', instance);
    },
    onModalClosed: (instance) => {
        console.log('Modal closed', instance);
    }
});
myModal1.open();
